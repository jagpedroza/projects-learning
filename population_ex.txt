SELECT count(*)
FROM countries
WHERE continent is 'Africa';

SELECT countries.continent, sum(population_years.population) as population_2005
FROM countries
INNER JOIN population_years
ON countries.id = population_years.country_id
WHERE countries.continent = 'Oceania' AND population_years.year = '2005';

SELECT countries.continent, avg(population_years.population) as AVGERAGE_POP
FROM countries
INNER JOIN population_years
ON countries.id = population_years.country_id
WHERE countries.continent = 'South America' AND population_years.year = '2003';

SELECT countries.name, population_years.population as minim_POP
FROM countries
INNER JOIN population_years
ON countries.id = population_years.country_id
WHERE population_years.year = '2007' and population_years.population is NOT NULL
ORDER BY population_years.population ASC
LIMIT 3;

SELECT avg(population_years.population) as AVG_POLAND
FROM countries
JOIN population_years
ON countries.id = population_years.country_id
WHERE countries.name = 'Poland';

SELECT count(countries.name)
FROM countries
WHERE countries.name like '%THE%';

SELECT countries.continent, avg(population_years.population) as AVG_POPULATION
FROM countries
JOIN population_years
ON countries.id = population_years.country_id
GROUP BY countries.continent
ORDER BY 2 DESC;
